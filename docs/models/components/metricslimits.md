# MetricsLimits

Date limits to get metrics.

## Example Usage

```typescript
import { MetricsLimits } from "@polar-sh/sdk/models/components";
import { RFCDate } from "@polar-sh/sdk/types";

let value: MetricsLimits = {
  minDate: new RFCDate("2024-11-25"),
  intervals: {
    hour: {
      maxDays: 844253,
    },
    day: {
      maxDays: 888924,
    },
    week: {
      maxDays: 905059,
    },
    month: {
      maxDays: 76555,
    },
    year: {
      maxDays: 857243,
    },
  },
};
```
