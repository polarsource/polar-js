# MetricsLimits

Date limits to get metrics.

## Example Usage

```typescript
import { MetricsLimits } from "@polar-sh/sdk/models/components";
import { RFCDate } from "@polar-sh/sdk/types";

let value: MetricsLimits = {
  minDate: new RFCDate("2022-11-24"),
  intervals: {
    hour: {
      maxDays: 265724,
    },
    day: {
      maxDays: 848972,
    },
    week: {
      maxDays: 859375,
    },
    month: {
      maxDays: 15344,
    },
    year: {
      maxDays: 719607,
    },
  },
};
```
